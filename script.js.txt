let current = 0;
const slides = document.querySelector('.slides');
const total = slides.children.length;

function showSlide(index) {
  slides.style.transform = `translateX(${-index * 100}%)`;
}

function nextSlide() {
  current = (current + 1) % total;
  showSlide(current);
}

setInterval(nextSlide, 4000);

}
